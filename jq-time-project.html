function convertTime() { const inputField = document.getElementById('inputMinutes'); const resultDiv = document.getElementById('result'); const input = inputField.value; const minutes = parseFloat(input); let resultText = ''; if (isNaN(minutes) || minutes < 0) { resultText = "â›” Please enter a valid number."; } else { let actualMinutes; if (minutes <= 2) { actualMinutes = 15; } else if (minutes <= 5) { actualMinutes = 25; } else if (minutes <= 10) { actualMinutes = 40; } else if (minutes <= 20) { actualMinutes = 60; } else if (minutes <= 30) { actualMinutes = 150; } else if (minutes <= 60) { actualMinutes = minutes * 3.5; } else { actualMinutes = minutes * 3; } const hours = Math.floor(actualMinutes / 60); const remainingMinutes = Math.round(actualMinutes % 60); if (hours > 0) { resultText = `${hours} hour(s) and ${remainingMinutes} minute(s)`; } else { resultText = `${remainingMinutes} minutes`; } } resultDiv.textContent = resultText; // Clear input and result after 10 seconds setTimeout(() => { inputField.value = ''; resultDiv.textContent = ''; }, 10000); }
